008public class Item
009{
010    private String aDescription;
011    private int aWeight;
012    private String aItemString;
013    private String aLongDescription;
014    private boolean aIsBeamer;
015    
016    
017    /**
018     * Constructeur naturel
019     * @param pDescription Description de l'item
020     * @param pWeight Poids de l'item
021     * @param pItemString Nom de l'item avec article
022     * @param pLDescription Description longue de l'item
023     * @param pIsBeamer true si l'Item est un Beamer, false sinon
024     */
025    public Item(final String pDescription, final int pWeight, final String pItemString, final String pLDescription, final boolean pIsBeamer){
026        this.aWeight = pWeight;
027        this.aDescription = pDescription;
028        this.aItemString = pItemString;
029        this.aLongDescription = pLDescription;
030        this.aIsBeamer = pIsBeamer;
031    }//Item(.,.,.,.)
032  
033    public String getDescription(){return this.aDescription;}
034    public int getWeight(){return this.aWeight;}
035    public String getItemString(){return this.aItemString;}
036    public String getLongDescription(){return this.aLongDescription;}
037    public boolean isBeamer(){return this.aIsBeamer;}
038    
039}
