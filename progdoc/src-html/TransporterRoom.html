001import java.util.ArrayList;
002/**
003 * Sous-classe de Room implémentant la Transporter Room
004 *
005 * @author Raphaël CANIN
006 * @version 2020
007 */
008public class TransporterRoom extends Room
009{
010    private ArrayList<Room> aRoomsList;
011    private Room aRoomTestMode;
012
013    /**
014     * Constructeur naturel
015     * @param pDescription String pour initialiser aDescription
016     * @param pImage String pour donner le nom de l'image souhaitée.
017     * @param pRoomsList Liste des Rooms à tirer au sort
018     */
019    public TransporterRoom(final String pDescription, final String pImage, final ArrayList<Room> pRoomsList)
020    {
021        super(pDescription, pImage);
022        this.aRoomsList = pRoomsList;
023    }//Room(.,.)
024
025    /**
026     * Renvoie la Room de sortie aléatoire
027     * @param pDirection Non important
028     * @return Room correspondant à la sortie demandée.
029     */
030    @Override public Room getExit (final String pDirection){
031        if (this.aRoomTestMode != null) return this.aRoomTestMode;        
032        return RoomRandomizer.findRandomRoom(this.aRoomsList);
033    }//getExit(.)  
034
035    /**
036     * Setter de aRoomNameTestMode
037     * @param pRoom Room à laquelle initialiser l'attribut
038     */
039    public void setRoomTestMode(final Room pRoom){
040        this.aRoomTestMode = pRoom;
041    }
042}
