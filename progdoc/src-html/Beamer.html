003 * Implémente le fonctionnement du beamer (exercice 7.44)
004 *
005 * @author Raphaël CANIN
006 * @version 2020
007 */
008public class Beamer extends Item
009{
010    private Room aChargedRoom = null;
011
012    /**
013     * Constructeur unique de la classe Beamer
014     */
015    public Beamer(){
016        super("beamer",400, "a beamer", " A beamer is a device that can be charged, and fired. When you charge the beamer, it memorizes the current room. When you fire the beamer, it transports you immediately back to the room it was charged in.",true);
017    }//Beamer()
018
019    /**
020     * Indique si le beamer est chargé ou non
021     * @return true si le beamer est chargé, false sinon.
022     */
023    public boolean isCharged(){
024        return (this.aChargedRoom != null);
025    }//isCharged()
026
027    /**
028     * Permet de charger le beamer
029     * @param pRoom Room à charger
030     */
031    public void chargeRoom(final Room pRoom){
032        this.aChargedRoom = pRoom;
033    }//chargeBeamer(.)   
034    
035    /**
036     * Permet de décharger le beamer
037     */
038    public void unchargeRoom(){
039        this.aChargedRoom = null;
040    }//unchargeBeamer()
041    
042    /**
043     * Renvoie la Room stockée dans le beamer
044     * @return la Room stockée dans le beamer
045     */
046    public Room fireRoom(){
047        return this.aChargedRoom;
048    }//fireBeamer()
049}
