001import java.util.Scanner;
002import java.io.File;
003import java.io.FileNotFoundException;
004/**
005 * Classe permettant la lecture de fichiers texte.
006 * Une partie de lecture() provient de M. Denis BUREAU.
007 * @author Raphaël CANIN
008 * @version 2020
009 */
010public class LectureFichier
011{
012    /**
013     * Lecture d'un fichier et renvoie de l'ensemble des lignes sous forme de String séparées
014     * par une virgule (pour utiliser split(",") ensuite)
015     * @param pNomFichier Nom du fichier à lire
016     * @return String du fichier lu.
017     */
018    public static String lecture( final String pNomFichier )
019    {
020        Scanner vSc;
021        StringBuilder vB = new StringBuilder("");
022        try {
023            vSc = new Scanner( new File( pNomFichier ) );
024            while ( vSc.hasNextLine() ) {
025                vB.append(vSc.nextLine() + ",");
026            }
027            return vB.toString();
028        } 
029        catch ( final FileNotFoundException pFNFE ) {
030            return null;
031        } 
032    } // lecture
033} // LectureFichier
