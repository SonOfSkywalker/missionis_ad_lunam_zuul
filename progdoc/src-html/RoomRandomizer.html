001import java.util.Random;
002import java.util.ArrayList;
003
004/**
005 * Classe renvoyant aléatoirement un Room du jeu.
006 *
007 * @author Raphaël CANIN
008 * @version 2020
009 */
010public abstract class RoomRandomizer
011{
012    public static Room findRandomRoom(final ArrayList<Room> pRoomsList) {
013      Random vGen = new Random();
014      return pRoomsList.get(vGen.nextInt(pRoomsList.size()));
015    }
016}
