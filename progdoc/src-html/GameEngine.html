192    /**
193     * Appelle la bonne méthode en fonction de la commande entrée en paramètre
194     * @param pC Command commande entrée par l'utilisateur
195     */
196    public void interpretCommand(final String pC)
197    {
198        this.aGui.println( "> " + pC );
199        Command vCommand = this.aParser.getCommand( pC );
200
201        if (this.aNbMoves >= MAXMOVES-1) lose();
202
203        this.aNbMoves +=1;
204
205        if (vCommand.isUnknown()) 
206        {
207            this.aGui.println( "I don't know what you mean..." );
208            this.aNbMoves += -1;
209            return;
210        }
211
212        switch (vCommand.getCommandWord()) {  
213            case "quit":        this.quit(vCommand);    break;  
214            case "go":          this.goRoom(vCommand);  break; 
215            case "help":        this.printHelp();       break;
216            case "look":        this.look(vCommand);    break;
217            case "eat":         this.eat(vCommand);     break;
218            case "back":        this.back(vCommand);    break;
219            case "test":        this.test(vCommand);    break;
220            case "take":        this.take(vCommand);    break;
221            case "drop":        this.drop(vCommand);    break;
222            case "inventory":   this.inventory();       break;
223            case "charge":      this.charge(vCommand);  break;
224            case "fire":        this.fire(vCommand);    break;
225            case "alea":        this.alea(vCommand);    break;
226
227            default: this.aGui.println("An error has occured.");
228        }  
229
230    }//interpretCommand(.)

244    /**
245     * Exécution de "go"
246     * @param pDirection Command pour la direction souhaitée
247     */
248    private void goRoom(final Command pDirection)
249    {
250        if (!(pDirection.hasSecondWord())) 
251        {
252            this.aGui.println( "Go where?" );
253            return;
254        }
255
256        Room vNextRoom = null;
257
258        String vDirectionWord = pDirection.getSecondWord();
259
260        vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirectionWord);  
261
262        //Cas spéciaux
263        if (vNextRoom == null) this.aGui.println("There is no door!");
264
265        else if (vNextRoom == Room.UNKNOWN_DIRECTION) this.aGui.println("Unknown direction!");
266        // Fin des cas spéciaux
267
268        else
269        {
270            if (this.aPlayer.getCurrentRoom().canGoThrough(vDirectionWord)) {            
271                this.aPlayer.walk(vNextRoom);
272            }
273            else {
274                this.aGui.println(this.aPlayer.lockedDoorProcess(vDirectionWord));
275            }
276            this.printLocationInfo();
277        }
278    }//goRoom(.)

298     * @param pC Command commande entrée par l'utilisateur commençant par look

300    private void look(final Command pC){
301        if (pC.getSecondWord()==null){
302            this.printLocationInfo();
303        }
304        else {
305            if (this.aPlayer.getCurrentRoom().getItem(pC.getSecondWord()) != null) {
306                this.aGui.println(this.aPlayer.getCurrentRoom().getItem(pC.getSecondWord()).getLongDescription());
307            }
308            else this.aGui.println("There is no such item in the room.");
309
310        }
311    }//look(.)

331    private void back(final Command pCommand){
332        if (pCommand.hasSecondWord()) this.aGui.println("The back command doesn't accept a second word.");
333        else if (this.aPlayer.cantGoBack()) this.aGui.println("You can't go back.");
334        else
335        {
336            this.aPlayer.walkback();

338        }

351    /**
352     * Procédure de test
353     * @param pC commande
354     */
355    private void test(final Command pC){
356        if(!pC.hasSecondWord()) 
357            this.aGui.println("Which file do you want to use for the test ?");
358
359        else{
360            LectureFichier vLecture = new LectureFichier();
361            String vCommandes = vLecture.lecture(pC.getSecondWord()+".txt");
362            if (vCommandes==null) this.aGui.println("This file doesn't exist, try again.");
363            else {
364                String[] vTab = vCommandes.split(",");
365                for (String vCommand : vTab) {
366                    this.interpretCommand(vCommand);
367                }
368            }
369
370        }
371    }

429    /**
430     * Procédure exécutant la commande alea
431     * @param pC Command commençant par alea
432     */
433    private void alea(final Command pC){
434        if (this.aTestMode) {
435            TransporterRoom vTR = (TransporterRoom) this.aAllRooms.get("TransporterRoom");   
436            if (pC.hasSecondWord() && (this.aAllRooms.get(pC.getSecondWord()) != null) ) {                       
437                vTR.setRoomTestMode(this.aAllRooms.get(pC.getSecondWord()));
438                this.aGui.println("The alea room has been set to " + pC.getSecondWord() + ".\n");
439            }
440            else {
441                vTR.setRoomTestMode(null);
442                this.aGui.println("The alea room has been deleted.");
443            }
444        }
445        else this.aGui.println("You need to be in test mode to use this command");
446    }//alea(.)
