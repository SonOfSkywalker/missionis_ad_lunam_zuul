001import java.util.HashMap;
002import java.util.Set;
003import java.util.Iterator;
004
005/**
006 * Classe Room servant à créer des salles
007 * @author Raphaël CANIN
008 * @version 2020.10.10
009 */
010public class Room
011{
012    // ### Attributs ###
013    private String aDescription;
014    private HashMap<String,Room> aExits;
015    private HashMap<String,Item> aItems = new HashMap<String,Item>() ;
016    private ItemList aItemListRoom = new ItemList();
017    private HashMap<String,Door> aDoors = new HashMap<String,Door>();
018
019    //D'après une idée de Baudouin Cortes, création de UNKNOWN_DIRECTION
020    public static final Room UNKNOWN_DIRECTION = new Room("Unknown Direction");
021
022    //Création d'une HashMap contenant toutes les directions possibles
023    private static HashMap<String,String> sDirections = new HashMap<String,String>();
024
025    //Image
026    private String aImageName;
027
028    // ### Constructeurs ###
029    /**
030     * Constructeur naturel
031     * @param pDescription String pour initialiser aDescription
032     * @param pImage String pour donner le nom de l'image souhaitée.
033     */
034    public Room(final String pDescription, final String pImage)
035    {
036        this.aDescription = pDescription;
037        this.aExits = new HashMap<String,Room>();
038        this.aImageName = "Images/" + pImage;
039    }//Room(.,.)
040
041    /**
042     * Constructeur sans image
043     * @param pDescription String pour initialiser aDescription
044     */
045    public Room(final String pDescription)
046    {
047        this.aDescription = pDescription;
048        this.aExits = new HashMap<String,Room>();
049    }//Room(.)
050
051    // ### Accesseurs ###
052    /**
053     * Accesseur de la description d'une Room
054     * @return La description de la salle
055     */
056    public String getDescription() { return this.aDescription;}//getDescription()
057
058    /**
059     * Accesseur de l'image d'une Room
060     * @return Le String contenant le nom de l'image
061     */
062    public String getImageName(){return this.aImageName;}
063    // ### Autres Méthodes ####
064
065    /**
066     * Positionnement d'une sortie + Ajout de la direction entrée dans la HashMap sDirections
067     * @param pS String Direction de la sortie
068     * @param pR Room Pièce à affecter à cette sortie
069     */
070    public void setExit (final String pS, final Room pR)
071    {
072        if (pR != null) {
073            this.sDirections.put(pS,"");
074            this.aExits.put(pS,pR);
075        }
076    }//setExit(.,.)
077
078    /**
079     * Renvoie la Room de sortie correspondant à la direction entrée. 
080     * Si la sortie ne contient pas de Room, on renvoie null. 
081     * Si le String entré ne correspond à aucune direction connue, on renvoie Room.UNKNOWN_DIRECTION
082     * @param pDirection String Direction souhaitée
083     * @return Room correspondant à la sortie demandée.
084     */
085    public Room getExit (final String pDirection){
086        if (sDirections.containsKey(pDirection)) return this.aExits.get(pDirection);
087        else return Room.UNKNOWN_DIRECTION;
088    }//getExit(.)
089
090    /**
091     * Renvoie un String détaillant toutes les sorties disponibles
092     * @return Une description des sorties disponibles.
093     */
094    public String getExitString(){
095        StringBuilder vExit = new StringBuilder(""); 
096        Set<String> keys = this.aExits.keySet();
097
098        for(String exit : keys) {
099            vExit.append(exit + " ");   
100        }
101
102        return vExit.toString();
103    }//getExitString()
104
105    /**
106     * Renvoie la description longue d'une Room :
107     *      You are in the Sea of Vapors
108     *      Exits : north
109     *      No item here.
110     *  @return Une description de la Room, ses sorties et les items présents
111     */
112    public String getLongDescription(){
113        String vItemsString = "";
114        if (this.aItemListRoom.isEmpty()) vItemsString = "No item here.";
115        else vItemsString = "Items in the room : " + this.aItemListRoom.getItemsString();
116        return "You are " + this.getDescription() + ".\n" + "Exits : " + this.getExitString() + "\n"  + vItemsString + "\n";
117    }//getLongDescription()
118
119    /**
120     * Permet de définir un item pour une Room
121     * @param pItem item à ajouter à la Room
122     */
123    public void addItem(final Item pItem){
124        this.aItemListRoom.addItem(pItem);
125    }//setItem(.)
126
127    /**
128     * Permet de retirer un item d'une Room
129     * @param pItemString item à retirer de la Room
130     */
131    public void removeItem(final String pItemString){
132        this.aItemListRoom.removeItem(pItemString);
133    }//removeItem(.)
134
135    /**
136     * Permet de récupérer l'item de la pièce
137     * @param pS String contenant le nom d'un item
138     * @return Renvoie l'item de la HashMap correspondant à la clé passée en paramètre, null sinon
139     */
140    public Item getItem(final String pS){
141        return this.aItemListRoom.getItem(pS);
142    }//getItem(.)
143
144    /**
145     * Informe si la Room passée en paramètre est une sortie de la salle courante
146     * @param pR Room à vérifier
147     * @return boolean true si la Room est une sortie, false sinon
148     */
149    public boolean isExit(final Room pR){
150        return this.aExits.containsValue(pR);
151    }
152
153    /**
154     * Positionnement d'une sortie + Ajout de la direction entrée dans la HashMap sDirections
155     * @param pS Direction de la sortie
156     * @param pD Door à affecter à cette sortie
157     */
158    public void setDoor (final String pS, final Door pD)
159    {
160        if (pD != null) {
161            this.aDoors.put(pS,pD);
162        }
163    }//setDoor(.,.)
164
165    /**
166     * @param pDirection Direction à étudier
167     * @return true si la direction entrée est accessible (non bloquée par une porte verouillée), false sinon
168     */
169    public boolean canGoThrough(final String pDirection) {
170        return this.aDoors.get(pDirection)==null || (this.aDoors.get(pDirection)!=null && !this.aDoors.get(pDirection).isLocked());
171    }
172
173    /**
174     * @param pDirection Direction à étudier
175     * @return Le nom de l'Item pouvant ouvrir la porte
176     */
177    public String getKeyString(final String pDirection){
178        return this.aDoors.get(pDirection).getKeyName();
179    }
180
181    /**
182     * @param pDirection Direction à étudier
183     * @return La Door correspond à la sortie demandée, null sinon.
184     */
185    public Door getExitDoor(final String pDirection){
186        return this.aDoors.get(pDirection);
187    }
188} // Room
