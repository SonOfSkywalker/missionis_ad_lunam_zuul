001import java.util.HashMap;
002import java.util.Stack;
003/**
004 * Classe permettant la création de joueurs dans le jeu. 
005 * Elle contient les méthodes propres au joueur.
006 * @author Raphael CANIN
007 * @version 2020
008 */
009public class Player
010{
011    private Room aCurrentRoom;
012    private Stack<Room> aBackRooms = new Stack<Room>();
013
014    private ItemList aInventory = new ItemList();
015    private int aMaxWeight = 900;
016
017    /**
018     * Contructeur de Player
019     * @param pR Room de départ
020     */
021    public Player(final Room pR){
022        this.aCurrentRoom = pR;
023    }//Player(.)
024
025    /**
026     * @return aCurrentRoom
027     * getter de aCurrentRoom
028     */
029    public Room getCurrentRoom(){return this.aCurrentRoom;}//getCurrentRoom()
030
031    /**
032     * @param pR
033     * Setter de aCurrentRoom
034     */
035    public void setCurrentRoom(final Room pR) {this.aCurrentRoom = pR;}//setCurrentRoom(.)
036
037    /**
038     * @param pNextRoom Room de destination
039     * Permet au joueur de se déplacer dans la bonne pièce
040     */
041    public void walk(final Room pNextRoom){
042        this.aBackRooms.push(this.aCurrentRoom);
043        this.setCurrentRoom(pNextRoom);
044    }//walk(.)
045
046    /**
047     * Retour dans la pièce précédente
048     */
049    public void walkback(){
050        this.aCurrentRoom = this.aBackRooms.pop();   
051    }//walkback()
052
053    /**
054     * On vérifie s'il reste des pièces stockées pour revenir en arrière
055     * @return boolean false s'il reste des pièces dans la pile ou si la Room au sommet de 
056     * la pile n'est pas une sortie de la Room actuelle, true sinon
057     */
058    public boolean cantGoBack(){
059        return this.aBackRooms.isEmpty() || !this.aCurrentRoom.isExit(this.aBackRooms.peek());
060    }//cantGoBack()
061
062    /**
063     * Prendre un Item
064     * @param pItemName Nom de l'Item à ramasser
065     * @return Message à afficher
066     */
067    public String takeItem(final String pItemName){
068        Item vItem = this.aCurrentRoom.getItem(pItemName);
069
070        if (vItem==null) return ("There is no such item in this room.");
071
072        if (this.aInventory.getTotalWeight() + vItem.getWeight() <= this.aMaxWeight) {
073            this.aInventory.addItem(vItem);
074            this.aCurrentRoom.removeItem(pItemName);
075
076            return ("You just took " + vItem.getItemString() + " .\n" + this.getInventoryInfo());}
077
078        return ("Maximum weight reached, you can't take this item.");
079    }//takeItem(.)
080
081    /**
082     * Déposer un Item
083     * @param pItemName Nom de l'Item à déposer
084     * @return Message à afficher au joueur
085     */
086    public String dropItem(final String pItemName){
087        Item vItem = this.aInventory.getItem(pItemName);
088
089        if (vItem==null) return ("There is no such item in your inventory.");
090
091        this.aCurrentRoom.addItem(vItem);
092        this.aInventory.removeItem(pItemName);
093
094        return ("You just dropped " + vItem.getItemString() + " .\n" + this.getInventoryInfo());
095    }//dropItem(.)
096
097    /**
098     * Infos sur l'inventaire
099     * @return String contenant les infos sur l'inventaire
100     */
101    public String getInventoryInfo(){
102        String vItemsString = "";
103        if (this.aInventory.isEmpty()) vItemsString = "You don't have any item in your inventory.";
104        else vItemsString = "\nItems in your inventory : \n" + this.aInventory.getItemsString();
105        return vItemsString + "\nCarried weight : " + this.aInventory.getTotalWeight() + "/" + this.aMaxWeight + "\n";
106    }
107
108    /**
109     * Permet de manger un cookie et d'en appliquer les effets
110     * @return String d'information au joueur
111     */
112    public String eatCookie(){
113        if (this.aInventory.getItem("cookie") != null) {
114            this.aMaxWeight += 900;
115            this.aInventory.removeItem("cookie");
116            return "You just ate a magic cookie. \nYour maximum carriable weight was set to 1900.\n" + this.getInventoryInfo();
117        }
118        else return "You don't have any cookie in your inventory";
119    }
120
121    /**
122     * Permet de charger un Beamer
123     * @param pS Nom du beamer
124     * @return Sortie textuelle pour afficher au joueur
125     */
126    public String chargeBeamer(final String pS){
127        Item vItem = this.aInventory.getItem(pS);
128        if (vItem==null) return "There is no such item in your inventory.";
129        else if (vItem!=null && vItem.isBeamer()) {
130            Beamer vBeamer = (Beamer) vItem;
131            vBeamer.chargeRoom(this.aCurrentRoom);
132            return "The current room has been charged in your beamer.";
133        }
134        else return "You can't do that.";
135    }
136
137    /**
138     * Permet de fire un Beamer
139     * @param pS Nom du beamer
140     * @return Sortie textuelle pour afficher au joueur
141     */
142    public String fireBeamer(final String pS){
143        Item vItem = this.aInventory.getItem(pS);
144        if (vItem==null) return "There is no such item in your inventory.\n";
145        else if (vItem!=null && vItem.isBeamer()) {
146            Beamer vBeamer = (Beamer) vItem;
147            if (vBeamer.isCharged()) {
148                this.setCurrentRoom(vBeamer.fireRoom());
149                this.aBackRooms.clear();
150                this.aInventory.removeItem(pS);
151                return "You have been teleported to the room charged in the beamer.\n";
152            }
153            else return "You need to charge the beamer first.\n";
154        }
155        else return "You can't do that.";
156    }
157
158    public String lockedDoorProcess(final String pDirection){
159        Item vKeyItem = this.aInventory.getItem(this.aCurrentRoom.getKeyString(pDirection));
160        if (vKeyItem != null) {            
161            this.aCurrentRoom.getExitDoor(pDirection).unlockDoor();
162            this.aInventory.removeItem(this.aCurrentRoom.getKeyString(pDirection));
163            return "The door was unlocked using " + vKeyItem.getDescription() +", you can now go through it.";
164        }
165        return "You need '" + this.aCurrentRoom.getKeyString(pDirection) + "' to unlock this door.";
166    }
167}
