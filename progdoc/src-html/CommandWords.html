001 /**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author  Michael Kolling and David J. Barnes + D.Bureau + Raphael CANIN
009 * @version 2008.03.30 + 2019.09.25 + 2020.09.22
010 */
011public class CommandWords
012{
013    // a constant array that will hold all valid command words
014    private final String[] aValidCommands = {"go","help","quit","look","eat","back","test","take","drop","inventory", "charge", "fire", "alea"};
015
016    /**
017     * Constructor - initialise the command words.
018     */
019    public CommandWords()
020    {
021        //
022    } // CommandWords()
023
024    /**
025     * Check whether a given String is a valid command word. 
026     * @param pString Commande à valider ou non
027     * @return true if a given string is a valid command,
028     * false if it isn't.
029     */
030    public boolean isCommand( final String pString )
031    {
032        for ( int i=0; i<this.aValidCommands.length; i++ ) {
033            if ( this.aValidCommands[i].equals( pString ) )
034                return true;
035        } // for
036        // if we get here, the string was not found in the commands
037        return false;
038    } // isCommand()
039    
040    /**
041     * Renvoie les différentes commandes disponibles sous forme de String
042     * @return String contenant toutes les commandes disponibles
043     */
044    public String getCommandList(){
045        StringBuilder vS = new StringBuilder("");
046        for (String vCommand : this.aValidCommands) {
047            vS.append(vCommand + " ");
048        }
049        return vS.toString();
050    }//getCommandList()
051   
052    
053} // CommandWords
